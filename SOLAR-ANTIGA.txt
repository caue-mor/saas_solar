                          <div className="flex items-center space-x-3 mb-2">
                            <h3 className="text-lg font-semibold text-gray-900">{contact.nome}</h3>
                            <Badge color={contact.status_leads_fotovoltaico?.cor}>
                              {contact.status_leads_fotovoltaico?.nome}
                            </Badge>
                          </div>
                          
                          <div className="space-y-1">
                            <p className="text-sm text-gray-600 flex items-center space-x-2">
                              <svg className="w-4 h-4" fill="none" stroke="currentColor" viewBox="0 0 24 24">
                                <path strokeLinecap="round" strokeLinejoin="round" strokeWidth={2} d="M3 5a2 2 0 012-2h3.28a1 1 0 01.948.684l1.498 4.493a1 1 0 01-.502 1.21l-2.257 1.13a11.042 11.042 0 005.516 5.516l1.13-2.257a1 1 0 011.21-.502l4.493 1.498a1 1 0 01.684.949V19a2 2 0 01-2 2h-1C9.716 21 3 14.284 3 6V5z" />
                              </svg>
                              <span>{contact.celular.replace('@s.whatsapp.net', '').replace('55', '')}</span>
                            </p>
                            
                            {contact.potencia_consumo_medio && (
                              <p className="text-sm text-gray-600 flex items-center space-x-2">
                                <svg className="w-4 h-4" fill="none" stroke="currentColor" viewBox="0 0 24 24">
                                  <path strokeLinecap="round" strokeLinejoin="round" strokeWidth={2} d="M13 10V3L4 14h7v7l9-11h-7z" />
                                </svg>
                                <span>Potência: {contact.potencia_consumo_medio}</span>
                              </p>
                            )}
                            
                            <p className="text-xs text-gray-400 flex items-center space-x-2">
                              <svg className="w-3 h-3" fill="none" stroke="currentColor" viewBox="0 0 24 24">
                                <path strokeLinecap="round" strokeLinejoin="round" strokeWidth={2} d="M12 8v4l3 3m6-3a9 9 0 11-18 0 9 9 0 0118 0z" />
                              </svg>
                              <span>Criado em: {formatDate(contact.created_on)}</span>
                            </p>
                          </div>
                        </div>

                        {/* Status e controles */}
                        <div className="flex flex-col sm:items-end space-y-3">
                          {/* Toggle atendimento automático */}
                          <div className="flex items-center space-x-3 bg-gray-50 rounded-lg p-3">
                            <div className="text-center">
                              <p className="text-xs font-medium text-gray-700 mb-1">Atendimento IA</p>
                              <div className="flex items-center space-x-2">
                                <span className="text-xs text-gray-500">
                                  {contact.atendimento_automatico ? 'Ativo' : 'Inativo'}
                                </span>
                                <Toggle
                                  checked={contact.atendimento_automatico}
                                  onChange={() => toggleAutoAttendance(contact.id, contact.atendimento_automatico)}
                                  disabled={!hasAIFeatures(user?.produto_plano) || user?.status_plano !== 'ativo'}
                                />
                              </div>
                            </div>
                          </div>

                          {/* Botões de ação */}
                          <div className="flex space-x-2">
                            <Button
                              onClick={() => handleEditContact(contact)}
                              variant="outline"
                              size="sm"
                            >
                              Editar
                            </Button>
                            <Button
                              onClick={() => handleDeleteContact(contact.id)}
                              variant="destructive"
                              size="sm"
                            >
                              Excluir
                            </Button>
                          </div>
                        </div>
                      </div>
                      
                      {/* Observações */}
                      {contact.observacoes_status && (
                        <div className="mt-4 p-3 bg-blue-50 border border-blue-200 rounded-lg">
                          <p className="text-sm text-blue-700">
                            <span className="font-medium">Observações:</span> {contact.observacoes_status}
                          </p>
                        </div>
                      )}
                    </div>
                  </div>
                </div>
              </Card>
            ))}
          </div>

          {filteredContacts.length === 0 && (
            <Card className="p-8 text-center">
              <div className="w-16 h-16 mx-auto mb-4 bg-gray-100 rounded-full flex items-center justify-center">
                <svg className="w-8 h-8 text-gray-400" fill="none" stroke="currentColor" viewBox="0 0 24 24">
                  <path strokeLinecap="round" strokeLinejoin="round" strokeWidth={2} d="M17 20h5v-2a3 3 0 00-5.356-1.857M17 20H7m10 0v-2c0-.656-.126-1.283-.356-1.857M7 20H2v-2a3 3 0 015.356-1.857M7 20v-2c0-.656.126-1.283.356-1.857m0 0a5.002 5.002 0 019.288 0M15 7a3 3 0 11-6 0 3 3 0 016 0zm6 3a2 2 0 11-4 0 2 2 0 014 0zM7 10a2 2 0 11-4 0 2 2 0 014 0z" />
                </svg>
              </div>
              <h3 className="text-lg font-medium text-gray-900 mb-2">Nenhum contato encontrado</h3>
              <p className="text-gray-500">Cadastre o primeiro contato da sua empresa.</p>
            </Card>
          )}
        </div>
      );

      // Aba do Kanban com filtros de tempo
      const renderKanbanTab = () => (
        <div className="h-full">
          <div className="flex flex-col lg:flex-row justify-between items-start lg:items-center mb-4 space-y-4 lg:space-y-0">
            <div>
              <h2 className="text-2xl sm:text-3xl font-bold text-gray-900">Kanban - Gestão Visual</h2>
              <p className="text-gray-600 mt-1">Organize e acompanhe seus contatos de forma visual</p>
            </div>
          </div>

          {/* Filtros do Kanban */}
          <Card className="p-4 mb-4">
            <div className="grid grid-cols-1 sm:grid-cols-2 lg:grid-cols-4 gap-4">
              <div className="relative">
                <input
                  type="text"
                  placeholder="Buscar contatos..."
                  value={kanbanSearchTerm}
                  onChange={(e) => setKanbanSearchTerm(e.target.value)}
                  className="pl-10 pr-4 py-2 border border-gray-300 rounded-lg focus:outline-none focus:ring-2 focus:ring-solar-500 focus:border-transparent w-full"
                />
                <svg className="absolute left-3 top-2.5 h-5 w-5 text-gray-400" fill="none" stroke="currentColor" viewBox="0 0 24 24">
                  <path strokeLinecap="round" strokeLinejoin="round" strokeWidth={2} d="M21 21l-6-6m2-5a7 7 0 11-14 0 7 7 0 0114 0z" />
                </svg>
              </div>

              <Select
                value={kanbanTimeFilter}
                onChange={setKanbanTimeFilter}
                label="Filtro de Tempo"
              >
                {Object.entries(TIME_FILTERS).map(([key, label]) => (
                  <option key={key} value={key}>{label}</option>
                ))}
              </Select>

              {kanbanTimeFilter === 'custom' && (
                <>
                  <Input
                    type="date"
                    value={kanbanCustomDateStart}
                    onChange={(e) => setKanbanCustomDateStart(e.target.value)}
                    label="Data inicial"
                  />
                  <Input
                    type="date"
                    value={kanbanCustomDateEnd}
                    onChange={(e) => setKanbanCustomDateEnd(e.target.value)}
                    label="Data final"
                  />
                </>
              )}

              {kanbanTimeFilter !== 'custom' && (
                <div className="text-sm text-gray-600 bg-gray-100 px-3 py-2 rounded-lg flex items-center">
                  <span className="font-medium">{filteredKanbanContacts.length}</span> contatos
                </div>
              )}
            </div>
          </Card>

          {/* Kanban Board */}
          <div className="bg-amber-50 rounded-xl p-4 shadow-inner border border-amber-200">
            <div className="flex space-x-4 overflow-x-auto pb-4">
              {statusLeads.map((status) => (
                <KanbanColumn
                  key={status.id}
                  status={status}
                  leads={contactsByStatus[status.id] || []}
                  onDrop={handleMoveContactToStatus}
                  onEdit={handleEditContact}
                />
              ))}
            </div>
          </div>

          {statusLeads.length === 0 && (
            <Card className="p-8 text-center mt-8">
              <div className="w-16 h-16 mx-auto mb-4 bg-gray-100 rounded-full flex items-center justify-center">
                <svg className="w-8 h-8 text-gray-400" fill="none" stroke="currentColor" viewBox="0 0 24 24">
                  <path strokeLinecap="round" strokeLinejoin="round" strokeWidth={2} d="M9 17V7m0 10a2 2 0 01-2 2H5a2 2 0 01-2-2V7a2 2 0 012-2h2a2 2 0 012 2m0 10a2 2 0 002 2h2a2 2 0 002-2M9 7a2 2 0 012-2h2a2 2 0 012 2m0 0V5a2 2 0 012-2h2a2 2 0 012 2v2M9 7h6" />
                </svg>
              </div>
              <h3 className="text-lg font-medium text-gray-900 mb-2">Nenhum status encontrado</h3>
              <p className="text-gray-500">Configure os status de leads para usar o Kanban.</p>
            </Card>
          )}
        </div>
      );

      return (
        <div>
          {message && (
            <div className="fixed top-4 right-4 z-50 w-96">
              <Alert type={messageType}>
                {message}
              </Alert>
            </div>
          )}
          
          {user ? renderDashboard() : renderLogin()}
          
          {/* Modal Sistema */}
          <Modal
            isOpen={showSystemDialog}
            onClose={() => {
              setShowSystemDialog(false);
              setEditingSystem(null);
              setSelectedFiles({imagem1: null, imagem2: null});
              setImagePreviews({imagem1: '', imagem2: ''});
            }}
            title={editingSystem ? 'Editar Sistema' : 'Cadastrar Sistema'}
            size="large"
          >
            <div className="space-y-6">
              <div className="grid grid-cols-1 sm:grid-cols-2 gap-6">
                <Select
                  value={systemForm.tipo_sistema}
                  onChange={(value) => setSystemForm({...systemForm, tipo_sistema: value})}
                  label="Tipo de Sistema *"
                >
                  <option value="">Selecione o tipo</option>
                  {getAvailableSystemTypes().map(type => (
                    <option key={type} value={type}>{type}</option>
                  ))}
                </Select>

                <Input
                  value={systemForm.potencia_usina}
                  onChange={(e) => setSystemForm({...systemForm, potencia_usina: e.target.value})}
                  placeholder="Ex: 5.5 kWp"
                  label="Potência da Usina"
                />
              </div>

              <Textarea
                value={systemForm.descricao}
                onChange={(e) => setSystemForm({...systemForm, descricao: e.target.value})}
                placeholder="Descreva o sistema fotovoltaico..."
                label="Descrição *"
                rows={3}
              />

              <div className="grid grid-cols-1 sm:grid-cols-2 gap-6">
                <Input
                  value={systemForm.economia_anual}
                  onChange={(e) => setSystemForm({...systemForm, economia_anual: e.target.value})}
                  placeholder="Ex: R$ 2.400,00"
                  label="Economia Anual"
                />

                <Input
                  value={systemForm.nome_cliente}
                  onChange={(e) => setSystemForm({...systemForm, nome_cliente: e.target.value})}
                  placeholder="Nome do cliente"
                  label="Nome do Cliente"
                />
              </div>

              <Textarea
                value={systemForm.detalhes}
                onChange={(e) => setSystemForm({...systemForm, detalhes: e.target.value})}
                placeholder="Detalhes técnicos, equipamentos utilizados..."
                label="Detalhes"
                rows={4}
              />

              {/* Upload de imagens moderno */}
              <div>
                <h4 className="text-lg font-medium text-gray-900 mb-4">Imagens do Sistema</h4>
                <div className="grid grid-cols-1 sm:grid-cols-2 gap-6">
                  {['imagem1', 'imagem2'].map((imageKey, index) => (
                    <div key={imageKey} className="space-y-3">
                      <label className="block text-sm font-medium text-gray-700">
                        Imagem {index + 1} {index === 0 && '(Principal)'}
                      </label>
                      
                      <div className="relative">
                        {(imagePreviews[imageKey] || systemForm[imageKey]) ? (
                          // Preview da imagem
                          <div className="relative group">
                            <div className="aspect-video w-full bg-gray-100 rounded-xl overflow-hidden border-2 border-dashed border-gray-300 hover:border-solar-400 transition-all duration-200">
                              <img
                                src={imagePreviews[imageKey] || systemForm[imageKey]}
                                alt={`Preview ${index + 1}`}
                                className="w-full h-full object-cover group-hover:scale-105 transition-transform duration-200"
                              />
                            </div>
                            
                            {/* Overlay com ações */}
                            <div className="absolute inset-0 bg-black bg-opacity-0 group-hover:bg-opacity-40 transition-all duration-200 flex items-center justify-center rounded-xl">
                              <div className="opacity-0 group-hover:opacity-100 transition-opacity duration-200 flex space-x-2">
                                <button
                                  type="button"
                                  onClick={() => {
                                    const input = document.getElementById(`file-${imageKey}`);
                                    input.click();
                                  }}
                                  className="bg-white text-gray-700 px-3 py-2 rounded-lg text-sm font-medium hover:bg-gray-50 shadow-lg transition-colors duration-200"
                                >
                                  <svg className="w-4 h-4 inline mr-1" fill="none" stroke="currentColor" viewBox="0 0 24 24">
                                    <path strokeLinecap="round" strokeLinejoin="round" strokeWidth={2} d="M4 16l4.586-4.586a2 2 0 012.828 0L16 16m-2-2l1.586-1.586a2 2 0 012.828 0L20 14m-6-6h.01M6 20h12a2 2 0 002-2V6a2 2 0 00-2-2H6a2 2 0 00-2 2v12a2 2 0 002 2z" />
                                  </svg>
                                  Alterar
                                </button>
                                <button
                                  type="button"
                                  onClick={() => removeSelectedImage(imageKey)}
                                  className="bg-red-500 text-white px-3 py-2 rounded-lg text-sm font-medium hover:bg-red-600 shadow-lg transition-colors duration-200"
                                >
                                  <svg className="w-4 h-4 inline mr-1" fill="none" stroke="currentColor" viewBox="0 0 24 24">
                                    <path strokeLinecap="round" strokeLinejoin="round" strokeWidth={2} d="M19 7l-.867 12.142A2 2 0 0116.138 21H7.862a2 2 0 01-1.995-1.858L5 7m5 4v6m4-6v6m1-10V4a1 1 0 00-1-1h-4a1 1 0 00-1 1v3M4 7h16" />
                                  </svg>
                                  Remover
                                </button>
                              </div>
                            </div>
                          </div>
                        ) : (
                          // Área de upload
                          <div
                            onClick={() => {
                              const input = document.getElementById(`file-${imageKey}`);
                              input.click();
                            }}
                            className="aspect-video w-full bg-gradient-to-br from-gray-50 to-gray-100 rounded-xl border-2 border-dashed border-gray-300 hover:border-solar-400 hover:bg-gradient-to-br hover:from-solar-50 hover:to-amber-50 transition-all duration-200 cursor-pointer group flex flex-col items-center justify-center p-6"
                          >
                            <div className="text-center">
                              <div className="w-12 h-12 mx-auto mb-4 bg-gradient-to-br from-solar-100 to-amber-100 rounded-full flex items-center justify-center group-hover:from-solar-200 group-hover:to-amber-200 transition-all duration-200">
                                <svg className="w-6 h-6 text-solar-600" fill="none" stroke="currentColor" viewBox="0 0 24 24">
                                  <path strokeLinecap="round" strokeLinejoin="round" strokeWidth={2} d="M4 16l4.586-4.586a2 2 0 012.828 0L16 16m-2-2l1.586-1.586a2 2 0 012.828 0L20 14m-6-6h.01M6 20h12a2 2 0 002-2V6a2 2 0 00-2-2H6a2 2 0 00-2 2v12a2 2 0 002 2z" />
                                </svg>
                              </div>
                              <h4 className="text-sm font-medium text-gray-900 mb-1 group-hover:text-solar-700 transition-colors duration-200">
                                Clique para adicionar imagem
                              </h4>
                              <p className="text-xs text-gray-500 group-hover:text-solar-600 transition-colors duration-200">
                                PNG, JPG até 5MB
                              </p>
                            </div>
                          </div>
                        )}
                        
                        {/* Input file oculto */}
                        <input
                          id={`file-${imageKey}`}
                          type="file"
                          accept="image/*"
                          onChange={(e) => handleFileSelect(e, imageKey)}
                          className="hidden"
                        />
                      </div>
                      
                      {/* Dica de uso */}
                      {index === 0 && (
                        <p className="text-xs text-gray-500 bg-blue-50 border border-blue-200 rounded-lg p-2">
                          💡 A primeira imagem será exibida como destaque nos cards
                        </p>
                      )}
                    </div>
                  ))}
                </div>
                
                {/* Informações sobre upload */}
                <div className="mt-4 p-4 bg-amber-50 border border-amber-200 rounded-lg">
                  <div className="flex items-start space-x-3">
                    <svg className="w-5 h-5 text-amber-600 mt-0.5 flex-shrink-0" fill="none" stroke="currentColor" viewBox="0 0 24 24">
                      <path strokeLinecap="round" strokeLinejoin="round" strokeWidth={2} d="M13 16h-1v-4h-1m1-4h.01M21 12a9 9 0 11-18 0 9 9 0 0118 0z" />
                    </svg>
                    <div>
                      <h5 className="text-sm font-medium text-amber-900 mb-1">Dicas para melhores resultados:</h5>
                      <ul className="text-xs text-amber-800 space-y-1">
                        <li>• Use imagens de alta qualidade (mínimo 800x600px)</li>
                        <li>• Prefira fotos tiradas durante o dia com boa iluminação</li>
                        <li>• Mostre diferentes ângulos do sistema instalado</li>
                        <li>• Inclua detalhes dos equipamentos quando possível</li>
                      </ul>
                    </div>
                  </div>
                </div>
              </div>

              <div className="flex justify-end space-x-4 pt-6 border-t">
                <Button
                  variant="outline"
                  onClick={() => {
                    setShowSystemDialog(false);
                    setEditingSystem(null);
                  }}
                >
                  Cancelar
                </Button>
                <Button onClick={handleSaveSystem} disabled={loading}>
                  {loading ? 'Salvando...' : (editingSystem ? 'Atualizar' : 'Cadastrar')}
                </Button>
              </div>
            </div>
          </Modal>

          {/* Modal Contato */}
          <Modal
            isOpen={showContactDialog}
            onClose={() => {
              setShowContactDialog(false);
              setEditingContact(null);
            }}
            title={editingContact ? 'Editar Contato' : 'Novo Contato'}
            size="medium"
          >
            <div className="space-y-6">
              <div className="grid grid-cols-1 sm:grid-cols-2 gap-6">
                <Input
                  value={contactForm.nome}
                  onChange={(e) => setContactForm({...contactForm, nome: e.target.value})}
                  placeholder="Nome completo"
                  label="Nome *"
                />

                <Input
                  value={contactForm.celular}
                  onChange={(e) => setContactForm({...contactForm, celular: e.target.value})}
                  placeholder="11999999999"
                  label="Celular *"
                />
              </div>

              <div className="grid grid-cols-1 sm:grid-cols-2 gap-6">
                <Input
                  value={contactForm.potencia_consumo_medio}
                  onChange={(e) => setContactForm({...contactForm, potencia_consumo_medio: e.target.value})}
                  placeholder="Ex: 8 kWp"
                  label="Potência de Consumo Médio"
                />

                <Select
                  value={contactForm.status_lead_id}
                  onChange={(value) => setContactForm({...contactForm, status_lead_id: parseInt(value)})}
                  label="Status do Lead"
                >
                  {statusLeads.map(status => (
                    <option key={status.id} value={status.id}>{status.nome}</option>
                  ))}
                </Select>
              </div>

              <div className="p-4 bg-gray-50 rounded-lg">
                <Toggle
                  checked={contactForm.atendimento_automatico}
                  onChange={(value) => setContactForm({...contactForm, atendimento_automatico: value})}
                  disabled={!hasAIFeatures(user?.produto_plano) || user?.status_plano !== 'ativo'}
                  label="Ativar atendimento automático"
                  description="Permite que a IA atenda automaticamente este contato"
                />
              </div>

              <Textarea
                value={contactForm.observacoes_status}
                onChange={(e) => setContactForm({...contactForm, observacoes_status: e.target.value})}
                placeholder="Observações sobre o contato..."
                label="Observações"
                rows={3}
              />

              <div className="flex justify-end space-x-4 pt-6 border-t">
                <Button
                  variant="outline"
                  onClick={() => {
                    setShowContactDialog(false);
                    setEditingContact(null);
                  }}
                >
                  Cancelar
                </Button>
                <Button onClick={handleSaveContact} disabled={loading}>
                  {loading ? 'Salvando...' : (editingContact ? 'Atualizar' : 'Cadastrar')}
                </Button>
              </div>
            </div>
          </Modal>
        </div>
      );
    };

    // Renderizar o sistema
    const root = ReactDOM.createRoot(document.getElementById('solar-system-root'));
    root.render(<SolarSystem />);
    </script>
</body>
